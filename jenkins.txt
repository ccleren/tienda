pipeline {
            agent {label 'iaw'}
                stages {
                        stage('Downloading configuration files') {
                            steps {
                                echo 'Descargando...'
                                sh 'rm -R .* || true'
                                sh 'rm -R * || true'
                                sh 'git clone https://github.com/ccleren/tienda.git .'
                            }
                        }
                        stage('Deploy docker') {
                            steps {
                                echo 'Contruyendo...'
                                sh 'docker-compose up -d'
                                
                            }
                        }
                        stage('Add PHP files') {
                            steps {
                                echo 'Añadiendo los ficheros PHP...'
                                sh 'cp PHP/* web/ || true'
                            }
                        }
                        stage('Add tables on iaw_db') {
                            steps {
                                echo 'Añadiendo tablas en iaw_db...'
                                sh 'docker exec -i tienda_bbdd sh -c "mysql -u root -psecret iaw_db < /bbdd/iaw_db.sql" || true'
                            }
                        }
                        stage('Install composer') {
                            steps {
                                echo 'Instalando las dependencias...'
                                sh 'cd web/ && composer install || true'
                                 }
                        }
                }
            post {
                always {
                    echo 'Running pipeline'
                }
                success {
                    echo 'Success'
                }
                failure {
                    echo 'Oops something went wrong'
                }
            }
        }
